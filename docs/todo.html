/etc/magic
proc run {file} {
    global mimetype

    if {[file executable $file]} {
        exec $file
        return
    }

    catch {set mimetype $mimetype(extension,[file extension $file])}

    if {![info exists mimetype]} {
        set mimetype $mimetype(magic,[exec /bin/file $file])
    }

    exec $mimetype(application,$mimetype) $file
}
